1-(valor 1,5 pontos) Quantos erros tem no script abaixo? Qual(is) são eles?
case "$file" in
    *.txt}
        echo "É um arquivo de texto."
        cat "$file"

    *.gif|*.jpg|*.png}
        echo "É uma figura."
        xloadimage "$file"

    *}
        echo "Não sei que arquivo é esse."

esac


Resposta 1
erros:
erro 1- falta ;; no final de cada case
erro 2- não se usa } o correto é )
erro 3- no inicio falta #!/bin/bash para informar que e um programa em bash
erro 4- e priciso ler o $file, seja por parametro ou interaçãp com o usuario
script correto:
#!/bin/bash

file=$1
case "$file" in
    *.txt)
        echo "É um arquivo de texto."
        cat "$file";;

    *.gif|*.jpg|*.png)
        echo "É uma figura."
        xloadimage "$file";;

    *)
        echo "Não sei que arquivo é esse.";;

esac
